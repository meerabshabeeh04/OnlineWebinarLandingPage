CREATE TABLE WEBINAR (
      WEBINAR_ID INT AUTO_INCREMENT PRIMARY KEY,
      TOPIC VARCHAR(250),
      DATE DATE,
      TIME VARCHAR(250),
      INTRODUCTION VARCHAR(5000),
      DURATION DOUBLE(10,2),
      IMAGE VARCHAR(250)
);

CREATE TABLE USER_ (
      USER_ID INT AUTO_INCREMENT PRIMARY KEY,
      EMAIL VARCHAR(250),
      GENDER VARCHAR(250),
      NAME VARCHAR(250),
      PH_NO BIGINT,
      QUALIFICATION VARCHAR(250)
);

CREATE TABLE WEBINAR_USER(
      USER_ID INT,
      WEBINAR_ID INT,
      CONSTRAINT PRIMARY KEY(USER_ID,WEBINAR_ID),
      CONSTRAINT FOREIGN KEY(USER_ID) REFERENCES USER_(USER_ID),
      CONSTRAINT FOREIGN KEY(WEBINAR_ID) REFERENCES WEBINAR(WEBINAR_ID)
);

CREATE TABLE WEBINAR_TOPICS(
      WEBINAR_TOPIC_ID INT AUTO_INCREMENT,
      WEBINAR_ID INT,
      TOPIC VARCHAR(250),
      DURATION VARCHAR(250),
      CONSTRAINT PRIMARY KEY(WEBINAR_TOPIC_ID,WEBINAR_ID,TOPIC),
      CONSTRAINT FOREIGN KEY(WEBINAR_ID) REFERENCES WEBINAR(WEBINAR_ID)
);
